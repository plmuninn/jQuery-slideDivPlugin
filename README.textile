A simple jQuery plugin to toogle hidden div contents.
<h1>Html Implementation:</h1>
<pre>
<code>
<div class="ClassName">
      <ul>
      <li><a href="#">link</a></li>
      <li><a href="#">link2</a></li>
      <li><a href="#">link3</a></li>
      </ul>
            <div class="sliderDiv-0">
              Content
            </div>
            <div class="sliderDiv-1">
              Content
            </div>
            <div class="sliderDiv-2">
              Content
            </div>
</div>
</code>
</pre>

<h1>JavaScript full implementation</h1>
<pre>
<code>
  $(".ClassName").slideDivs({
     'up'         :  10,        //Type ["slow","fast"] or time after to hide content
     'down'       : 'slow',     //Type or time after content show
     'class'      : 'sliderDiv',//Div content class name
     'first'      :  true,      //Are the first content div should be show after script initialization
     'doubleClick':  false,     //If you want to make second click of link to hide actual content for this link
     'hideChild'  :  false,     //If you want to hide child slideDivs implementation (IMPORTANT)Give this on parents not on children!
     'auto'       :  false,     //If content should hide automatic
     'autoTime'   :  5000       //Time to hide after click
  });
 </code>
 </pre>
<h1>Short JavaScript implementation</h1>
 <pre>
 <code>
  $(".ClassName").slideDivs();
 </code>
 </pre>
