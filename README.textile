<h1>Html Implementation:</h1>
<pre>
<code>
<div class="ClassName">
      <ul>
      <li><a href="#">link</a></li>
      <li><a href="#">link2</a></li>
      <li><a href="#">link3</a></li>
      </ul>
            <!--Content divs structure:-->
            <div class="sliderDiv-0">
              Content
            </div>
            <div class="sliderDiv-1">
              Content
            </div>
            <div class="sliderDiv-2">
              Content
            </div>
</div>
</code>
</pre>

<h1>Short JavaScript implementation</h1>
 <pre>
 <code>
  $(".ClassName").slideDivs();
 </code>
 </pre>


<h1>JavaScript full implementation</h1>
<pre>
<code>
  $(".ClassName").slideDivs({
     'up'             :  10,         //Type ["slow","fast"] or time after to hide content
     'down'           : 'slow',      //Type or time after content show
     'class'          : 'sliderDiv', //Div content class name
     'first'          :  true,       //Are the first content div should be show after script initialization
     'doubleClick'    :  false,      //If you want to make second click of link to hide actual content for this link
     'hideChild'      :  false,      //If you want to hide child slideDivs implementation (IMPORTANT)Give this on parents not on children!
     'auto'           :  false,      //If content should hide automatic
     'autoTime'       :  5000        //Time to hide after click
     'initFunction'   : 'click',     //Showing start event
     'startFunction'  : 'slideDown', //Show effect
     'endFunction'    : 'slideUp',   //Hide effect
     'hideFunction'   : 'hide',       //Automatic hide effect
              'ajaxValues'    : {                                   //Ajax object with configuration
                 'enabled'               : false,                   //If ajax is enabled
                 'type'                  : null,                    //Type of ajax method ( {null | loading},post,get,ajax), if {null | loading} you can show in a->href witch element should be show "index.html .body" else you should use loadedAjaxFunction
                 'automaticLoading'      : false,                   //If ajax should init after initFunction
                 'loadingImageSrc'       : 'img/ajax-loader.gif',   //Path to loadImage
                 'loadText'              : 'Loading',               //Loading text
                 'ajaxLang'              : 'php',                   //Which type of language
                 'ajaxLangV'             : 5,                       //Which version of language
                 'ajaxReturn'            : 'xml',                   //Return type value
                 'ajaxFunction'          : function(){},            //If you won't own ajax function (type must be ajax)
                 'loadedAjaxFunction'    : function(responseText, container, liElement){    //Method to transformation respond from ajax:
                                             return $("body",responseText);                 //container -> div where will be print respond
                                             }                                              //liElement -> li element when was activate ajax
                }
  });
 </code>
 </pre>
